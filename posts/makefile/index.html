基本结构

  • 基本语法

    target:dependencies
	command1
	command2
	...

  • 简单示例

    app:main.o utils.o
	gcc -o app main.o utils.o
main.o:main.c
	gcc -o main.o main.c
utils.o:utils.c
	gcc -o utils.o utils.c

clean:
	rm -rf *.o app

核心概念

  • 变量定义与使用

    CC = gcc
CFLAGS = -Wall -O2
TARGET = app
OBJS = main.o utils.o

$(TAEGET):$(OBJS
	$(CC) $(CFLAGS) -o $(TARGET) $(OBJS)

  • 自动变量

    $@ #当前规则的目标文件名
$< #第一个依赖文件名
$^ #所有依赖文件列表

  • 通配符与模式规则

    SRCS = $(wildcard *.c)
OBJS = $(patsubst %.c,%.o,$(SRCS))
%.o:%.c
	$(CC) $(CFLAGS) -c -o $@ $<

  • 伪目标

    .PHONY: clean install
clean:
	rm -f $(OBJS) $(TARGET)

install:
	cp $(TARGET) /usr/local/bin

  • 目录创建自动化

    $(BUILD_DIR)/%.o:$(SRC_DIR)/%.c
	@makedir -p $(@D) #自动创建目录
	$(CC) $(CFLAGS) -c $< -o $@
